具体来说，是在 `nova_repo/nova-block/src/components/novablock/NovaBlockEditor.tsx` 文件的 `handleAIAction` 函数里。因为大模型（Gemma）有时候就算被严格限制了，还是会顽固地在代码两端加上 markdown 的代码块标记（比如开头自带 ` ```python `，结尾自带 ` ``` `），导致直接插进编辑器里会多出这些无用的反引号。

所以我给它加了一个正则替换，把头尾的 markdown 标记给“咔嚓”剪掉了。

**修改前**：
```typescript
} else if (type === 'insert_code_block') {
  if (editor) {
    // 确保在插入前聚焦，并清理多余空行
    editor.chain().focus().insertContent({
      type: 'codeBlock',
      attrs: { language: attrs?.language || 'plain' },
      content: [{ type: 'text', text: value }] // 直接原样插入了 AI 吐出来的 value
    }).run();
  }
}
```

**修改后**：
```typescript
} else if (type === 'insert_code_block') {
  if (editor) {
    // 确保在插入前聚焦，并清理多余空行
    editor.chain().focus().insertContent({
      type: 'codeBlock',
      attrs: { language: attrs?.language || 'plain' },
      // 这里加了正则替换，把头尾的 ``` 和语言标识给去掉了！
      content: [{ type: 'text', text: value.replace(/^\s*```\w*\n?/, '').replace(/\n?```\s*$/, '') }] 
    }).run();
  }
}
```
