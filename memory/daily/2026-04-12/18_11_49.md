def install_ollama():
    base_dir = os.path.dirname(os.path.abspath(__file__))
    bin_dir = os.path.join(base_dir, "bin")
    ollama_exe = os.path.join(bin_dir, "ollama.exe")
    
    # 只要检测到引擎文件已经存在，就直接跳过下载！

        return
